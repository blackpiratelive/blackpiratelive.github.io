{{- $u := .Destination -}} {{- $isRemote := or (hasPrefix $u "http") (hasPrefix $u "https") -}} {{- $src := $u -}} {{- if not $isRemote -}} {{- $path := strings.TrimPrefix "/" $u -}} {{- $res := resources.Get $path -}} {{- if $res -}} {{- $src = $res.RelPermalink -}} {{- else -}} {{- $src = $u | relURL -}} {{- end -}} {{- end -}}
